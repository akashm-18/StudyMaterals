Create the student table (find the primary and foreign key)and
mark for their 5 subjects and display the student name if the student scored 100 in
any of the subject.

CREATE TABLE STUDENTS (
STUDENTID INT PRIMARY KEY,
STUDENTNAME VARCHAR(50) NOT NULL
);

CREATE TABLE MARKS (
MARKID INT PRIMARY KEY,
STUDENTID INT,
SUBJECTNAME VARCHAR(20) NOT NULL,
MARK INT,
FOREIGN KEY STUDENTID REFERENCES STUDENTS(STUDENTID)
);
