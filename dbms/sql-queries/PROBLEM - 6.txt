PROBLEM - 6

SQL query to insert, update, delete and retrieve data from table.

CREATE TABLE STUDENT (
STUDENTID INT PRIMARY KEY,
NAME VARCHAR(20) NOT NULL,
VALUE INT
)

INSERT INTO STUDENT(STUDENTID,NAME,VALUE)
(1,'AKASH',20),
(2,'ABISHEK',30)

UPDATE STUDENT SET VALUE = 40 
WHERE NAME = 'AKASH'

DELETE FROM STUDENT 
WHERE NAME = 'ABISHEK'

SELECT * FROM STUDENT;

SELECT NAME FROM STUDENT
WHERE VALUE = 40;
